### while in inserto mode...
^r = insert text from a register
^a = insert text from register '.'
^p = completion menu
^x = special "completion mode" submode of insert
- - ^] = tag
- - ^p = pull from previous context
- - ^n = pull from next context
- - ^f = file completion
- - ^l = line
- - ^o = omnicompletion
